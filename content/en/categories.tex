\section{Age Groups and Events}

\subsection{Age Groups}
Age groups are determined by year of birth (date: December 31, \eventYear).

\begin{itemize}
    \item \textbf{Group 1:} Born \eventYear\ and younger
    \item \textbf{Group 2:} Born 2024 - 2023
    \item \textbf{Group 3:} Born 2022 - 2021
    % Add more age groups as needed
\end{itemize}

\subsection{Competition Events}
\textbf{Individual Events:}
\begin{itemize}
    \item 50m Freestyle
    \item 100m Freestyle
    \item 200m Freestyle
    \item 50m Breaststroke
    \item 100m Breaststroke
    \item 50m Backstroke
    \item 100m Backstroke
    \item 50m Butterfly
    \item 100m Butterfly
    \item 200m Individual Medley
\end{itemize}

\textbf{Relay Events:}
\begin{itemize}
    \item 4 x 50m Freestyle Relay
    \item 4 x 50m Medley Relay
\end{itemize}

\subsection{Entry Limits}
Each participant may enter a maximum of 4 individual events and 2 relay events.
