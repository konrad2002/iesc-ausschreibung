% CUSTOM COMMANDS
% ============================================================================
% Highlighted text in main color
\newcommand{\highlight}[1]{\textcolor{maincolor}{\textbf{#1}}}

% Flag in top right corner (absolute positioning, no extra space)
\newcommand{\flagtopright}[1]{%
    \AddToShipoutPictureFG*{%
        \put(\LenToUnit{\paperwidth-1.4cm},\LenToUnit{\paperheight-1cm}){%
            \includegraphics[width=1.2cm]{images/#1}%
        }%
    }%
}

% Important notice box
\newcommand{\importantbox}[1]{%
    \vspace{0.3cm}
    \noindent\fcolorbox{maincolor}{white}{%
        \parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
            \textcolor{maincolor}{\textbf{Wichtig:}} #1
        }
    }
    \vspace{0.3cm}
}
